     # Função para verificar se um número é primo
    def is_prime(num):
        if num <= 1:
            return False
        for i in range(2, int(num ** 0.5) + 1):
            if num % i == 0:
                return False
        return True
    
    # Função para gerar números primos até o número limite
    def generate_primes(limit):
        primes = []
        for i in range(2, limit + 1):
            if is_prime(i):
                primes.append(i)
        return primes
    
    # Entrada do usuário
    try:
        limit = int(input("Digite o número limite para gerar números primos: "))
        primes = generate_primes(limit)
        print(f"Números primos até {limit}: {primes}")
    except ValueError:
        print("Por favor, insira um número válido.")
